

Deklaration 
`````c++
class MyClass {
public:
    MyClass(int size);
    MyClass(const MyClass& other); // copy ctor
    ~MyClass();
private:
    int* _data;
    int  _size;
};
`````

Implementering 
`````c++
MyClass::MyClass(const MyClass& other)
    : _size(other._size),
      _data(new int[other._size]) // ny allokering
{
    // kopiér indhold — ikke bare pointeren!
    for (int i = 0; i < _size; ++i)
        _data[i] = other._data[i];
}
`````

Main 
`````c++
MyClass a(3);
MyClass b = a;  // copy constructor
// a og b har SEPARATE arrays i hukommelsen
// ændrer du b, påvirker det ikke a
`````
