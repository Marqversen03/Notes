

Deklaration 
`````c++
class MyClass {
public:
    MyClass(int size);
    MyClass(const MyClass& other);
    MyClass& operator=(const MyClass& other);
    ~MyClass();
private:
    int* _data;
    int  _size;
};
`````

Implementering 
`````c++
MyClass& MyClass::operator=(const MyClass& other) {
    if (this == &other) return *this; // 1. self-assign
    delete[] _data;                    // 2. frigiv gammelt
    _size = other._size;               // 3. kopiér størrelse
    _data = new int[_size];           // 4. allokér nyt
    for (int i = 0; i < _size; ++i)   // 5. kopiér data
        _data[i] = other._data[i];
    return *this;                      // 6. returnér *this
}
`````

Main 
`````c++
MyClass a(3), b(5);
b = a;   // copy assignment — ikke constructor
b = b;   // self-assign — safe pga. guard
`````
